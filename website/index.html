⚡

ЧТО ЭТО?

PromeTorch — полностью независимый фреймворк глубокого обучения, созданный с нуля на C++ и CUDA. Это не форк, не обёртка, не копия PyTorch. Каждая строка кода написана заново.

🇷🇺

ЗАЧЕМ РОССИИ?

Технологическая независимость — вопрос национальной безопасности. Зависимость от иностранных библиотек в критической инфраструктуре (ИИ, оборона, медицина) недопустима. PromeTorch — шаг к суверенитету в области ИИ.

✓

ПОЧЕМУ НЕ КОПИЯ?

Копирование создаёт зависимость от оригинала. Мы изучили архитектуру PyTorch, но реализовали всё независимо: свой TensorImpl, свой Autograd Engine, свои CUDA-кернелы. Полный контроль над кодом.

# PromeTorch Python API
import promethorch as pt

# Создание тензоров
x = pt.randn([32, 784])
y = pt.randint(0, 10, [32])

# Нейронная сеть
model = pt.nn.Sequential(
    pt.nn.Linear(784, 256),
    pt.nn.ReLU(),
    pt.nn.Linear(256, 10)
)

# Оптимизатор
optimizer = pt.optim.Adam(
    model.parameters(),
    lr=0.001
)

# Training loop
for epoch in range(100):
    logits = model(x)
    loss = pt.nn.functional.cross_entropy(logits, y)

    optimizer.zero_grad()
    loss.backward()
    optimizer.step()

    print(f"Loss: {loss.item():.4f}")
// PromeTorch C++ API
#include <ATen/ATen.h>
#include <torch/nn/nn.h>
#include <torch/optim/optim.h>

int main() {
    // Создание тензоров
    auto x = torch::randn({32, 784});
    auto y = torch::randint(0, 10, {32});

    // Нейронная сеть
    auto linear1 = std::make_shared<nn::Linear>(784, 256);
    auto linear2 = std::make_shared<nn::Linear>(256, 10);

    // Оптимизатор
    optim::Adam optimizer(
        {linear1->parameters(), linear2->parameters()},
        optim::AdamOptions(0.001)
    );

    // Training loop
    for (int epoch = 0; epoch < 100; ++epoch) {
        auto h = linear1->forward(x).relu();
        auto logits = linear2->forward(h);
        auto loss = F::cross_entropy(logits, y);

        optimizer.zero_grad();
        loss.backward();
        optimizer.step();

        std::cout << "Loss: " << loss.item() << std::endl;
    }
}
